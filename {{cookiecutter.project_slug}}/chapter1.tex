\begin{table}[h]
  \centering
  \caption{Example table with data from CSV file.}
  \label{tab:example}
  \begin{tabular}{cc}
    \toprule
    Column A & Column B \\
    \midrule
    \csvreader[no head, column count=2]{data/table.csv}{1=\colA,2=\colB}%
    {\colA & \colB \\}
    \bottomrule
  \end{tabular}
\end{table}
